#
# SmokeRand battery with low RAM consumption (4 MiB or less).
# It is a simplified version of the `brief` battery with significantly smaller
# samples. It also doesn't include "collision over" tests and birthday spacings
# tests that work with 64-bit values.
#
# Warning: it contains a simplified version of gap test that doesn't detect
# PRNGs with 2^32 period (uses ngaps=2500000 instead of ngaps=10000000).
#

# Frequency tests
monobit_freq test=monobit_freq   nvalues=1048576  # 2^20
byte_freq    test=nbit_words_freq  bits_per_word=8 average_freq=256 nblocks=256
word16_freq  test=nbit_words_freq  bits_per_word=16 average_freq=32 nblocks=64

# Birthday spacings tests
bspace32_1d      test=bspace_nd nbits_per_dim=32 ndims=1 nsamples=1024 get_lower=1
bspace32_1d_high test=bspace_nd nbits_per_dim=32 ndims=1 nsamples=1024 get_lower=0
bspace8_4d       test=bspace_nd nbits_per_dim=8  ndims=4 nsamples=1024 get_lower=1
bspace4_8d       test=bspace_nd nbits_per_dim=4  ndims=8 nsamples=1024 get_lower=1
bspace4_8d_dec   test=bspace4_8d_decimated  step=4096 # 2^12

# Gap tests
gap_inv512   test=gap shl=9    ngaps=2500000

# Linear complexity tests
linearcomp_high test=linearcomp nbits=50000 bitpos=high
linearcomp_mid  test=linearcomp nbits=50000 bitpos=mid
linearcomp_low  test=linearcomp nbits=50000 bitpos=low

# Matrix rank tests
matrixrank_1024       test=matrixrank n=1024 max_nbits=64
matrixrank_1024_low8  test=matrixrank n=1024 max_nbits=8

# Mod3 test
mod3 test=mod3 nvalues=4194304 # 2^24
